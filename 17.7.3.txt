money = int(input('Введите сумму:'))
per_cent = {'ТКБ': 5.6, 'СКБ': 5.9, 'ВТБ': 4.28, 'СБЕР': 4.0}
deposit = []
for key in per_cent:deposit.append(float(per_cent[key]*money/100))
m_deposit = max(deposit)

print(deposit)
print('Максимальная сумма, которую вы можете заработать ' + str(m_deposit))
